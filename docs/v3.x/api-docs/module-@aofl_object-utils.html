Example
let user = {
  name: 'Alan',
  account: {
    active: true,
    products: {
      '1': true,
      '2': true,
      '3': true
    }
  },
  preferences: {
    locale: 'en-US'
  }
};

deepAssign(user, 'account.products', {
  2: false
});
{ // new ref
  name: 'Alan',
  account: { // new ref
    active: true,
    products: { // new ref
      '1': true,
      '2': false,
      '3': true
    }
  },
  preferences: { // same ref
    locale: 'en-US'
  }
};
